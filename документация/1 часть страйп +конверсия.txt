TIER_HIERARCHY = ['free', 'lite', 'pro', 'premium']

def require_tier(minimum_tier: str):
    def dependency(current_user = Depends(get_current_user)):
        user_tier_index = TIER_HIERARCHY.index(current_user.tier or 'free')
        required_index  = TIER_HIERARCHY.index(minimum_tier)
        if user_tier_index < required_index:
            raise HTTPException(
                status_code=403,
                detail={
                    "error": "tier_required",
                    "required": minimum_tier,
                    "current": current_user.tier,
                }
            )
        return current_user
    return dependency
```

        ('lite',    'monthly'): settings.STRIPE_PRICE_LITE_MONTHLY,
        ('lite',    'annual'):  settings.STRIPE_PRICE_LITE_ANNUAL,

        "metadata": {"user_id": str(user.id), "tier": tier},

    const { data, expiresAt } = JSON.parse(savedChart);
    if (Date.now() < expiresAt) {
        await api.saveAnonymousChart(data);  // POST /api/v1/chart/save-anonymous
        localStorage.removeItem('anonymous_chart');
    }
}
```
